       dev-down:

       migrate:

  <action>
    1. Create `.github/workflows/ci.yml` per D-09 and D-10:
       ```yaml
       name: CI

       on:
         push:
           branches: [main, new]
         pull_request:
           branches: [main]

       jobs:
         ci:
           runs-on: ubuntu-latest

           services:
             postgres:
               image: postgres:16-alpine
               env:
                 POSTGRES_PASSWORD: postgres
                 POSTGRES_DB: website_test
                 POSTGRES_USER: postgres
               options: >-
                 --health-cmd pg_isready
                 --health-interval 10s
                 --health-timeout 5s
                 --health-retries 5
               ports:
                 - 5432:5432

           env:
             DATABASE_URL: postgres://postgres:postgres@localhost:5432/website_test?sslmode=disable

           steps:
             - uses: actions/checkout@v4

             - uses: actions/setup-go@v5
               with:
                 go-version: '1.26'
                 cache: true

             - name: Lint
               uses: golangci/golangci-lint-action@v6
               with:
                 version: v1.61.0

             - name: Test
               run: go test ./... -v -race

             - name: Build
               run: go build -o bin/server ./cmd/server
       ```

       Key details per D-10: Single job, sequential lint -> test -> build. Conserves free-tier minutes.
       Key detail per D-09: Postgres service container with health check (Pitfall 5 -- ensures Postgres is ready before tests run).
       Go module caching via `actions/setup-go` with `cache: true`.
       golangci-lint pinned to v1.61.0 to match local (Pitfall 4).
       Triggers on push to main and new branches, and pull_request to main.
  </action>
