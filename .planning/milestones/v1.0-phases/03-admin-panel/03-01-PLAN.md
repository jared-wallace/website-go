    **RequireSession middleware** in `internal/middleware/auth.go`:
    ```go
    package middleware

    import (
        "net/http"
        "github.com/alexedwards/scs/v2"
    )

    func RequireSession(sm *scs.SessionManager) func(http.Handler) http.Handler {
        return func(next http.Handler) http.Handler {
            return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                if !sm.GetBool(r.Context(), "authenticated") {
                    http.Redirect(w, r, "/admin/login", http.StatusSeeOther)
                    return
                }
                next.ServeHTTP(w, r)
            })
        }
    }
    ```
    Returns a middleware closure that checks `sm.GetBool(r.Context(), "authenticated")`. If false, redirects to `/admin/login` with 303. If true, calls next.

    package main

    import (
        "fmt"

    func main() {
        if len(os.Args) < 2 {
